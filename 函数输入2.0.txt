            cachedCalculation(expression, ...args) {
                const key = `${expression}-${args.join('-')}`;
                if (this.calculationCache.has(key)) {
                    return this.calculationCache.get(key);
                }

                try {
                    const compiled = this.compileExpression(expression);
                    const scope = {};
                    const vars = ['x', 'y', 'z', 't', 'theta', 'u', 'v'];
                    args.forEach((arg, i) => {
                        if (i < vars.length) scope[vars[i]] = arg;
                    });
                    
                    const result = compiled.evaluate(scope);
                    this.calculationCache.set(key, result);
                    return result;
                } catch (error) {
                    return NaN;
                }
            }

            // 改进的自适应采样算法
            adaptiveSampling(fn, min, max, maxPoints = 1000, tolerance = 0.01) {
                const points = [];
                const step = (max - min) / 100;
                
                // 初始采样
                for (let x = min; x <= max; x += step) {
                    const y = fn(x);
                    if (!isNaN(y) && Math.abs(y) < 1e6) { // 过滤过大值
                        points.push({x, y});
                    }
                }
                
                if (points.length <= 2) return points;
                
                // 使用迭代版本的 Douglas-Peucker 算法
                return this.simplifyCurveIterative(points, tolerance, maxPoints);
            }
            
            simplifyCurveIterative(points, tolerance, maxPoints) {
                if (points.length <= maxPoints) return points;
                
                const stack = [{start: 0, end: points.length - 1}];
                const keep = new Set([0, points.length - 1]);
                
                while (stack.length > 0 && keep.size < maxPoints) {
                    const {start, end} = stack.pop();
                    
                    let maxDist = 0;
                    let maxIndex = start;
                    
                    for (let i = start + 1; i < end; i++) {
                        const dist = this.perpendicularDistance(points[i], points[start], points[end]);
                        if (dist > maxDist) {
                            maxDist = dist;
                            maxIndex = i;
                        }
                    }
                    
                    if (maxDist > tolerance) {
                        keep.add(maxIndex);
                        if (maxIndex - start > 1) stack.push({start, end: maxIndex});
                        if (end - maxIndex > 1) stack.push({start: maxIndex, end});
                    }
                }
                
                // 按顺序返回保留的点
                return points.filter((_, i) => keep.has(i));
            }
            
            perpendicularDistance(point, lineStart, lineEnd) {
                const dx = lineEnd.x - lineStart.x;
                const dy = lineEnd.y - lineStart.y;
                const lengthSquared = dx * dx + dy * dy;
                
                if (lengthSquared === 0) {
                    return Math.sqrt(
                        Math.pow(point.x - lineStart.x, 2) + 
                        Math.pow(point.y - lineStart.y, 2)
                    );
                }
                
                const t = Math.max(0, Math.min(1, 
                    ((point.x - lineStart.x) * dx + (point.y - lineStart.y) * dy) / lengthSquared
                ));
                
                const projX = lineStart.x + t * dx;
                const projY = lineStart.y + t * dy;
                
                return Math.sqrt(
                    Math.pow(point.x - projX, 2) + 
                    Math.pow(point.y - projY, 2)
                );
            }

                this.currentMode = mode;
                this.currentView = mode;
                
                document.querySelectorAll('.tab').forEach(tab => {
                    tab.classList.remove('active');
                });
                document.querySelector(`.tab[data-mode="${mode}"]`).classList.add('active');
                
                document.getElementById('input-2d').style.display = mode === '2d' ? 'block' : 'none';
                document.getElementById('input-3d').style.display = mode === '3d' ? 'block' : 'none';
                
                document.querySelectorAll('.view-btn').forEach(btn => {
                    btn.classList.remove('active');
                });
                document.querySelector(`.view-btn[data-view="${mode}"]`).classList.add('active');
                
                this.updateUI();
                this.clearPlot();
            }

            async plotFunction() {
                if (this.isPlotting || !this.checkLibraries()) return;
                
                const startTime = performance.now();
                this.showLoading();
                this.hideGraphError();

                try {
                    // 验证输入
                    if (!this.validateInputs()) {
                        throw new Error('输入验证失败，请检查输入');
                    }
                    
                    if (this.currentMode === '2d') {
                        await this.plot2DFunction();
                    } else {
                        await this.plot3DFunction();
                    }

                    const endTime = performance.now();
                    this.lastPlotTime = endTime - startTime;
                    
                    this.analyzeFunctionProperties();
                    this.addAssistantMessage('函数图像绘制完成。点击"函数性质"按钮查看详细分析。', true);

                } catch (error) {
                    console.error('绘图错误:', error);
                    this.showGraphError(`绘图失败: ${error.message}`);
                    this.addAssistantMessage(`绘图过程中出现了错误：${error.message}`, true);
                } finally {
                    this.hideLoading();
                }
            }
            
            validateInputs() {
                // 验证坐标范围
                const xMin = parseFloat(document.getElementById('x-min').value);
                const xMax = parseFloat(document.getElementById('x-max').value);
                const yMin = parseFloat(document.getElementById('y-min').value);
                const yMax = parseFloat(document.getElementById('y-max').value);
                
                if (isNaN(xMin) || isNaN(xMax) || isNaN(yMin) || isNaN(yMax)) {
                    this.addAssistantMessage('坐标范围输入无效，请检查数字格式', true);
                    return false;
                }
                
                if (xMin >= xMax || yMin >= yMax) {
                    this.addAssistantMessage('坐标范围最小值必须小于最大值', true);
                    return false;
                }
                
                // 验证当前模式下的表达式
                if (this.currentMode === '2d') {
                    const type = document.getElementById('function-type-2d').value;
                    let expression = '';
                    
                    if (type === 'explicit') {
                        expression = document.getElementById('function-expr').value;
                    } else if (type === 'implicit') {
                        expression = document.getElementById('implicit-expr').value;
                    } else if (type === 'polar') {
                        expression = document.getElementById('polar-expr').value;
                    } else if (type === 'parametric') {
                        const xExpr = document.getElementById('param-x').value;
                        const yExpr = document.getElementById('param-y').value;
                        if (!xExpr || !yExpr) {
                            this.addAssistantMessage('参数方程需要输入x(t)和y(t)', true);
                            return false;
                        }
                        expression = xExpr; // 只验证一个
                    }
                    
                    if (!expression || expression.trim() === '') {
                        this.addAssistantMessage('请输入函数表达式', true);
                        return false;
                    }
                    
                    try {
                        this.compileExpression(expression);
                    } catch (e) {
                        this.addAssistantMessage(`表达式错误: ${e.message}`, true);
                        return false;
                    }
                }
                
                return true;
            }

            async plot2DFunction() {

                    const yValues = points.map(p => p.y);

                    this.lastPointCount = xValues.length;

                    traces.push({
                        x: xValues,
                        y: yValues,
                        type: 'scattergl',
                        mode: 'lines',
                        line: {color: '#64ffda', width: 3},

                    });

                    const yExpr = document.getElementById('param-y').value;

                        x: xValues,
                        y: yValues,
                        type: 'scattergl',
                        mode: 'lines',
                        line: {color: '#ff6b6b', width: 3},
                        name: `参数方程`

            analyzeExplicitFunction(expression) {
                // 使用数学分析库进行更精确的分析
                let domain = '实数域';
                let range = '实数域';
                let parity = '无法确定';
                let periodicity = '非周期';
                let monotonicity = '无法确定';
                
                try {
                    // 定义域分析
                    if (expression.includes('log(') || expression.includes('ln(')) {
                        domain = 'x > 0';
                    } else if (expression.includes('sqrt(') || expression.includes('√')) {
                        domain = '被开方数 ≥ 0';
                    } else if (expression.includes('/x') || expression.includes('1/x')) {
                        domain = 'x ≠ 0';
                    } else if (expression.includes('1/(')) {
                        domain = '分母 ≠ 0';
                    }
                    
                    // 奇偶性分析（简化版）
                    const testNegative = expression.replace(/x/g, '(-x)');
                    const simplified = math.simplify(expression);
                    const simplifiedNeg = math.simplify(testNegative);
                    
                    if (math.simplify(`${simplified} - (${simplifiedNeg})`).toString() === '0') {
                        parity = '偶函数';
                    } else if (math.simplify(`${simplified} + (${simplifiedNeg})`).toString() === '0') {
                        parity = '奇函数';
                    } else {
                        parity = '非奇非偶';
                    }
                    
                    // 周期性检测
                    if (expression.includes('sin(') || expression.includes('cos(') || 
                        expression.includes('tan(') || expression.includes('cot(')) {
                        periodicity = '周期函数';
                    }
                    
                    // 单调性简化分析
                    if (expression.includes('x^2') && !expression.includes('x^3')) {
                        monotonicity = 'x<0递减，x>0递增';
                    } else if (expression === 'x') {
                        monotonicity = '严格递增';
                    } else if (expression === '-x') {
                        monotonicity = '严格递减';
                    } else if (expression === 'exp(x)') {
                        monotonicity = '严格递增';
                    } else if (expression === 'log(x)') {
                        monotonicity = '严格递增';
                    }
                    
                } catch (error) {
                    console.warn('自动分析失败，使用简化分析:', error);
                }
                
                this.updatePropertiesPanel(
                    `y = ${expression}`,
                    domain,
                    range,
                    parity,
                    periodicity,
                    monotonicity
                );
            }

                document.getElementById('property-domain').textContent = domain;

                document.getElementById('property-parity').textContent = parity;

                document.getElementById('property-monotonicity').textContent = monotonicity;

            // 发送消息到数学助手
            async sendMessage() {
                const input = document.getElementById('chat-input');
                const message = input.value.trim();
                if (!message) return;

                // 添加用户消息
                this.addAssistantMessage(message, false);
                input.value = '';
                
                // 禁用发送按钮
                const sendBtn = document.getElementById('send-btn');
                sendBtn.disabled = true;

                // 显示输入指示器
                const typingIndicator = this.showTypingIndicator();

                try {
                    // 生成助手响应
                    const response = this.generateAssistantResponse(message);
                    
                    // 模拟网络延迟
                    await new Promise(resolve => setTimeout(resolve, 500 + Math.random() * 1000));
                    
                    this.hideTypingIndicator();
                    this.addAssistantMessage(response, true);
                } catch (error) {
                    this.hideTypingIndicator();
                    this.addAssistantMessage('抱歉，处理您的消息时出现了问题。', true);
                } finally {
                    sendBtn.disabled = false;
                }
            }

            // 生成助手响应（基于当前函数和问题）
            generateAssistantResponse(message) {
                const lowerMessage = message.toLowerCase();
                const currentExpr = this.currentExpression;
                
                // 如果还没有绘制函数
                if (!currentExpr || currentExpr === '未定义') {
                    if (lowerMessage.includes('性质') || lowerMessage.includes('特征') ||

        // 初始化应用 - 修复版本
        let visualizer;

        function initApplication() {
            // 双重检查库是否加载
            if (typeof Plotly === 'undefined' || typeof math === 'undefined') {
                console.error('Required libraries not loaded, retrying...');
                // 延迟后重试
                setTimeout(initApplication, 500);
                return;
            }
            
            try {
                visualizer = new MathVisualizer();
                
                // 设置初始函数
                const exprInput = document.getElementById('function-expr');
                if (exprInput) {
                    exprInput.value = 'sin(x)';
                }
                
                // 初始绘制 - 确保DOM完全就绪
                if (visualizer && visualizer.plotFunction) {
                    setTimeout(() => {
                        try {
                            visualizer.plotFunction();
                        } catch(e) {
                            console.error('Initial plot failed:', e);
                        }
                    }, 300);
                }
            } catch(e) {
                console.error('Initialization failed:', e);
            }
        }

        // 库加载检查 - 改进版
        function checkAndInit() {
            if (typeof Plotly !== 'undefined' && typeof math !== 'undefined') {
                initApplication();
                return true;
            }
            return false;
        }
        
        // 方法1: 如果libsLoaded已设置
        if (window.libsLoaded && window.libsLoaded.plotly && window.libsLoaded.mathjs) {
            setTimeout(initApplication, 100);
        } 
        // 方法2: 如果有错误
        else if (window.libsError && (window.libsError.plotly || window.libsError.mathjs)) {
            console.warn('Library load errors detected:', window.libsError);
        }
        // 方法3: 等待DOM加载后检查
        else {
            if (document.readyState === 'loading') {
                document.addEventListener('DOMContentLoaded', function() {
                    // 给库加载额外时间
                    setTimeout(checkAndInit, 1000);
                });
            } else {
                // DOM已加载，等待库
                setTimeout(checkAndInit, 1000);
            }
        }
        
        // 备用初始化 - 作为最后手段
        window.addEventListener('load', function() {
            setTimeout(checkAndInit, 2000);
        });
    </script>
