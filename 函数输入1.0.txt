        class MathVisualizer {

            }

            setupEventListeners() {

                }, 1000);
            }

            // 防抖绘图调度
            schedulePlot() {

                }

                }, 300); // 300ms防抖

            // 预编译数学表达式
            compileExpression(expression) {
                if (this.compiledExpressions.has(expression)) {
                    return this.compiledExpressions.get(expression);
                }

                try {
                    const compiled = math.compile(expression);
                    this.compiledExpressions.set(expression, compiled);
                    return compiled;
                } catch (error) {
                    console.error('表达式编译错误:', error);
                    return null;
                }
            }

            // 带缓存的数学计算
            cachedCalculation(expression, x, y = null, z = null) {
                const key = `${expression}-${x}-${y}-${z}`;
                if (this.calculationCache.has(key)) {
                    return this.calculationCache.get(key);
                }

                try {
                    const compiled = this.compileExpression(expression);
                    if (!compiled) return NaN;

                    const scope = y !== null ? (z !== null ? {x, y, z} : {x, y}) : {x};
                    const result = compiled.evaluate(scope);
                    this.calculationCache.set(key, result);
                    return result;
                } catch (error) {
                    return NaN;
                }
            }

            // 自适应采样算法
            adaptiveSampling(fn, min, max, maxPoints = 1000) {
                const points = [];
                const step = (max - min) / 100;
                
                let x = min;
                while (x <= max) {
                    const y = fn(x);
                    if (!isNaN(y)) {
                        points.push({x, y});
                    }
                    x += step;
                }

                // 简化曲线（Douglas-Peucker算法简化版）
                return this.simplifyCurve(points, 0.01);
            }

            simplifyCurve(points, tolerance) {
                if (points.length <= 2) return points;

                let maxDistance = 0;
                let index = 0;
                const start = points[0];
                const end = points[points.length - 1];

                for (let i = 1; i < points.length - 1; i++) {
                    const distance = this.perpendicularDistance(points[i], start, end);
                    if (distance > maxDistance) {
                        maxDistance = distance;
                        index = i;
                    }
                }

                if (maxDistance > tolerance) {
                    const left = this.simplifyCurve(points.slice(0, index + 1), tolerance);
                    const right = this.simplifyCurve(points.slice(index), tolerance);
                    return left.slice(0, -1).concat(right);
                } else {
                    return [start, end];
                }
            }

            perpendicularDistance(point, lineStart, lineEnd) {
                const area = Math.abs(
                    (lineEnd.x - lineStart.x) * (lineStart.y - point.y) - 
                    (lineStart.x - point.x) * (lineEnd.y - lineStart.y)
                );
                const lineLength = Math.sqrt(
                    Math.pow(lineEnd.x - lineStart.x, 2) + Math.pow(lineEnd.y - lineStart.y, 2)
                );
                return lineLength > 0 ? area / lineLength : 0;
            }

            async plot2DFunction() {
                const xMin = parseFloat(document.getElementById('x-min').value);
                const xMax = parseFloat(document.getElementById('x-max').value);

                    const points = this.adaptiveSampling(fn, xMin, xMax, 1000);

                    const xValues = points.map(p => p.x);
                    const yValues = points.map(p => p.y);

                    this.lastPointCount = xValues.length;

                    traces.push({
                        x: xValues,
                        y: yValues,
                        type: 'scattergl',
                        mode: 'lines',
                        line: {color: '#64ffda', width: 3},
                        name: `y = ${expression}`
                    });
                } else if (type === 'parametric') {

                    const tMax = parseFloat(document.getElementById('param-t-max').value);

                    if (!xExpr || !yExpr) return;

                    this.currentExpression = `x = ${xExpr}, y = ${yExpr}`;
                    
                    const tValues = math.range(tMin, tMax, (tMax - tMin) / 500).toArray();
                    const xValues = tValues.map(t => this.cachedCalculation(xExpr, t));
                    const yValues = tValues.map(t => this.cachedCalculation(yExpr, t));

                };

                    displayModeBar: true,
                    displaylogo: false,
                    responsive: true,
                    scrollZoom: true

            // AI助教功能
            addAIMessage(message, isAI = true) {
                const chatContainer = document.getElementById('chat-container');
                const messageDiv = document.createElement('div');
                messageDiv.className = `message ${isAI ? 'ai' : 'user'}`;
                messageDiv.textContent = message;
                chatContainer.appendChild(messageDiv);
                chatContainer.scrollTop = chatContainer.scrollHeight;
            }

            showTypingIndicator() {
                const chatContainer = document.getElementById('chat-container');
                const typingDiv = document.createElement('div');
                typingDiv.className = 'message ai typing';
                typingDiv.id = 'typing-indicator';
                for (let i = 0; i < 3; i++) {
                    const dot = document.createElement('div');
                    dot.className = 'typing-dot';
                    typingDiv.appendChild(dot);
                }
                chatContainer.appendChild(typingDiv);
                chatContainer.scrollTop = chatContainer.scrollHeight;
                return typingDiv;
            }

            hideTypingIndicator() {
                const typingIndicator = document.getElementById('typing-indicator');
                if (typingIndicator) {
                    typingIndicator.remove();
                }
            }

            // 发送消息到AI助教（模拟）
            async sendMessage() {
                const input = document.getElementById('chat-input');
                const message = input.value.trim();
                if (!message) return;

                // 添加用户消息
                this.addAIMessage(message, false);
                input.value = '';

                // 显示输入指示器
                const typingIndicator = this.showTypingIndicator();

                // 模拟AI响应（实际应用中应调用AI API）
                setTimeout(() => {
                    this.hideTypingIndicator();
                    
                    // 根据消息内容生成响应
                    let response = this.generateAIResponse(message);
                    this.addAIMessage(response, true);
                }, 1000 + Math.random() * 2000);
            }

            // 生成AI响应（模拟）
            generateAIResponse(message) {
                const lowerMessage = message.toLowerCase();
                
                if (lowerMessage.includes('性质') || lowerMessage.includes('特征')) {
                    return `函数 ${this.currentExpression || '当前'} 的性质分析：
- 定义域：实数域（除非有分母为零或负数开偶次方的情况）
- 值域：取决于具体函数形式
- 奇偶性：需要进一步分析函数对称性
- 单调性：通过导数分析函数的增减区间`;
                } else if (lowerMessage.includes('图像') || lowerMessage.includes('图形')) {
                    return `函数图像分析：
- 图像形状：${this.currentExpression ? '已绘制在右侧' : '请先绘制函数图像'}
- 关键点：包括零点、极值点、拐点等
- 渐近线：分析函数在无穷远处的行为
- 对称性：观察图像是否关于原点或y轴对称`;
                } else if (lowerMessage.includes('应用') || lowerMessage.includes('用途')) {
                    return `函数的实际应用：
- 数学建模：描述自然现象和工程问题
- 物理学：运动学、波动等领域的数学模型
- 经济学：成本、收益、需求等函数关系
- 工程学：信号处理、控制系统设计等`;
                } else if (lowerMessage.includes('导数') || lowerMessage.includes('微分')) {
                    return `导数与微分：
- 导数表示函数在某点的变化率
- 微分是函数变化的线性近似
- 通过导数可以分析函数的单调性和极值`;
                } else if (lowerMessage.includes('积分')) {
                    return `积分与应用：
- 积分是导数的逆运算
- 用于计算面积、体积、位移等
- 定积分表示函数在区间上的累积效应`;
                } else {
                    return `关于您的问题"${message}"，我可以帮助您分析函数性质、解释图像特征、讨论数学概念和应用。请告诉我您想了解的具体内容，或尝试点击下方的快捷问题按钮。`;
                }
            }
